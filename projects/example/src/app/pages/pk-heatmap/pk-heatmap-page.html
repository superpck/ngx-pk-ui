pk-heatmap

GitHub-style contribution heatmap. Pass an array of {{ '{' }} date, no, holiday? {{ '}' }} objects. Supports 4 color schemes, optional legend, and a custom tooltip formatter. The optional holiday field appends a holiday name to the tooltip.

Usage

Basic

import {{ '{' }} PkHeatmap {{ '}' }} from 'ngx-pk-ui';
import type {{ '{' }} PkHeatmapDay {{ '}' }} from 'ngx-pk-ui';

@Component({{ '{' }}
  imports: [PkHeatmap],
  template: `
    <pk-heatmap [data]="data" colorScheme="green" />
  `,
{{ '}' }})
export class MyComponent {{ '{' }}
  data: PkHeatmapDay[] = [
    {{ '{' }} date: '2026-05-01', no: 4 {{ '}' }},
    {{ '{' }} date: '2026-05-02', no: 7 {{ '}' }},
    {{ '{' }} date: '2026-05-03', no: 1 {{ '}' }},
    // ...
  ];
{{ '}' }}

PkHeatmapDay interface

interface PkHeatmapDay {{ '{' }}
  date:     string | Date;  // ISO string or Date object
  no:       number;         // contribution count (0 = empty cell)
  holiday?: string;         // optional — appended to tooltip e.g. "— วันแรงงาน"
{{ '}' }}

// Example with holidays:
data: PkHeatmapDay[] = [
  {{ '{' }} date: '2026-05-01', no: 0,  holiday: 'วันแรงงาน' {{ '}' }},
  {{ '{' }} date: '2026-05-04', no: 3,  holiday: 'วันฉัตรมงคล' {{ '}' }},
  {{ '{' }} date: '2026-05-05', no: 7 {{ '}' }},
  // ...
];

// Tooltip will show: "3 contributions on Mon, 4 May 2026 — วันฉัตรมงคล"

Custom tooltip formatter

readonly fmt = (day: PkHeatmapDay): string =>
  `${{ '{' }}day.no{{ '}' }} commits on ${{ '{' }}new Date(day.date).toLocaleDateString(){{ '}' }}`;

// template:
// <pk-heatmap [data]="data" [tooltipFormat]="fmt" />

Inputs

Input Type Default Description
data ★ PkHeatmapDay[] required Array of {{ '{' }} date, no, holiday? {{ '}' }} — holiday (optional) appends a label to the tooltip
startDate Date | string | null 1 year ago Start of the displayed range
endDate Date | string | null today End of the displayed range
colorScheme 'green' | 'blue' | 'purple' | 'orange' 'green' Cell color palette
locale PkLocale 'en' Locale for day/month labels and default tooltip date format — 17 locales supported (e.g. 'th', 'en', 'ja')
showLegend boolean true Show Less / More legend row
showTooltip boolean true Show tooltip on cell hover
tooltipFormat (day: PkHeatmapDay) => string null Custom tooltip text factory; default: "3 contributions on Fri, 16 May 2026"
dayLabels boolean true Show Mon / Wed / Fri labels on the left
monthLabels boolean true Show month names above the grid

★ required. Level intensity is calculated using quartile of all non-zero values (p25 → L1, p50 → L2, p75 → L3, above → L4) — prevents outliers from washing out all other cells.
