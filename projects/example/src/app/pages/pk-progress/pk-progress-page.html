pk-progress

Line and circle progress indicators with status, striped mode, animation, and percentage display.

Interactive demo

Quick presets

Usage

import { PkProgressComponent } from 'ngx-pk-ui';
import type { ProgressConfig } from 'ngx-pk-ui';

@Component({
  imports: [PkProgressComponent],
})
export class AnyPage {
  progressConfig: ProgressConfig = {
    type: 'line',
    percent: 65,
    status: 'normal',
    showInfo: true,
    strokeWidth: 8,
  };
}

<pk-progress [config]="progressConfig" />

API

InputTypeDefaultDescription

config.percentnumber0Progress percentage (0-100)
config.status'normal' | 'success' | 'error' | 'warning''normal'Status color/style variant
config.showInfobooleantrueShow percent/icon text
config.strokeWidthnumber8Line height or circle stroke width
config.labelstring—Optional label text (line mode)
config.colorstring—Custom progress color override
config.stripedbooleanfalseEnable striped pattern (line mode)
config.animatedbooleanfalseAnimate stripe movement (line mode)
config.indeterminatebooleanfalseIndeterminate animation (line mode)
