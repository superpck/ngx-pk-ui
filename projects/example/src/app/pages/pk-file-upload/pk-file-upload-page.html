pk-file-upload

File upload with drag & drop and browser-native thumbnail preview (image, PDF, text). No extra libraries required.

Images only (max 2 MB)

Accept only image files, limit size to 2 MB per file.

Any file type — with simulated upload

Add image, PDF, text, or other files then click Upload files to simulate an HTTP request (1.5 s delay).

Upload with HTTP request

Template

<pk-file-upload
  #uploader
  accept="image/*,.pdf"
  [multiple]="true"
  [maxSize]="10_000_000"
  [uploading]="isUploading()"
  (onUpload)="handleUpload($event)"
/>

Component

import { Component, inject, signal, viewChild } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { PkFileUpload, type PkUploadFile } from 'ngx-pk-ui';

@Component({
  imports: [PkFileUpload],
  template: `...`,
})
export class MyComponent {
  private http = inject(HttpClient);
  readonly uploader = viewChild<PkFileUpload>('uploader');
  isUploading = signal(false);

  async handleUpload(files: PkUploadFile[]): Promise<void> {
    this.isUploading.set(true);
    const formData = new FormData();
    files.map(f => f.file).forEach(f => formData.append('files', f));
    await this.http.post('/api/upload', formData).toPromise();
    this.isUploading.set(false);
    this.uploader()?.clear();  // reset after success
  }
}

React to every add / remove

<pk-file-upload
  [multiple]="true"
  [maxFiles]="5"
  (filesChange)="onFilesChange($event)"
/>
onFilesChange(files: PkUploadFile[]): void {
  // fires on every add/remove — files includes both valid and errored entries
  const valid = files.filter(f => !f.error);
  console.log(valid.length, 'ready to upload');
}

PkUploadFile interface

interface PkUploadFile {
  id: number;                          // unique auto-increment id
  file: File;                          // original File object
  previewType: 'image'|'pdf'|'text'|'none';
  previewUrl: string | null;           // blob: URL for image/pdf, null otherwise
  textContent: string | null;          // first 300 chars for text files
  error: string | null;                // set when file fails validation
}
