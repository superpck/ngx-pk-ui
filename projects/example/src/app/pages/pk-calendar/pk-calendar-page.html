pk-calendar

Modern calendar with Year / Month / Week / Day / Agenda views, built-in event form, drag & drop, multi-day event bars, and TH/EN locale support.

Demo

Click a date to create  ·  Click an event to edit  ·  Drag events to reschedule

Installation

npm install ngx-pk-ui

Usage

import { PkCalendar } from 'ngx-pk-ui';
import type { PkCalendarEvent, PkCalendarView, PkEventMoveResult } from 'ngx-pk-ui';

@Component({
  imports: [PkCalendar],
})
export class MyComponent {
  events: PkCalendarEvent[] = [
    {
      id: 1,
      title: 'Team Meeting',
      type: 'meeting',        // meeting | appointment | birthday | holiday
                               // festival | event | task | reminder | other
      start: new Date(),
      end: new Date(Date.now() + 3_600_000),
      priority: 'high',       // low | medium | high | urgent
      location: 'Room A',
      description: 'Weekly sync.',
      attachments: [{ label: 'Agenda', url: 'https://docs.google.com/...' }],
      canEdit: true,
      canDelete: true,
    },
    {
      id: 2,
      title: 'Songkran Holiday',
      type: 'holiday',
      start: new Date('2026-04-13'),
      end: new Date('2026-04-15'),
      allDay: true,
      color: '#10b981',
      canEdit: false,
      canDelete: false,
    },
  ];

  onCreate(ev: PkCalendarEvent) {
    this.events = [...this.events, { ...ev, id: Date.now() }];
  }
  onUpdate(ev: PkCalendarEvent) {
    this.events = this.events.map(e => e.id === ev.id ? ev : e);
  }
  onDelete(ev: PkCalendarEvent) {
    this.events = this.events.filter(e => e.id !== ev.id);
  }
  onMove(result: PkEventMoveResult) {
    this.events = this.events.map(e =>
      e.id === result.event.id
        ? { ...e, start: result.newStart, end: result.newEnd ?? e.end }
        : e
    );
  }
}
<pk-calendar
  [events]="events"
  locale="EN"
  startOfWeek="monday"
  (onEventCreate)="onCreate($event)"
  (onEventUpdate)="onUpdate($event)"
  (onEventDelete)="onDelete($event)"
  (onEventMove)="onMove($event)"
/>

CSS custom properties

pk-calendar {
  --pk-cal-primary:    #4f46e5;  /* accent colour */
  --pk-cal-today-bg:   #eff6ff;  /* today cell background */
  --pk-cal-border:     #e5e7eb;  /* grid border */
  --pk-cal-row-height: 120px;    /* month view cell min-height */
}

API — PkCalendar inputs

InputTypeDefaultDescription
eventsPkCalendarEvent[][]Events to display
view'year'|'month'|'week'|'day'|'agenda''month'Active calendar view
currentDateDatenew Date()Initial navigation date
locale'EN'|'TH''EN'Language for labels and month names
startOfWeek'sunday'|'monday''sunday'First day of the week
readonlybooleanfalseDisable all create / edit actions
showWeekNumbersbooleanfalseShow week number column in month view

API — PkCalendar outputs

OutputPayloadDescription
(onEventCreate)PkCalendarEventUser submitted the new-event form
(onEventUpdate)PkCalendarEventUser saved changes to an existing event
(onEventDelete)PkCalendarEventUser deleted an event
(onEventMove)PkEventMoveResultEvent was dragged to a new date
(onEventClick)PkCalendarEventUser clicked an event chip
(onDateClick)DateUser clicked an empty day cell
(onViewChange)PkCalendarViewUser switched the view
(onNavigate)DateUser navigated prev / next / today
