Installation

// angular.json
"styles": ["node_modules/ngx-pk-ui/styles/pk-table.css"]

// or include everything
"styles": ["node_modules/ngx-pk-ui/styles/pk-ui.css"]

// or CSS import
@import 'ngx-pk-ui/styles/pk-table.css';

Usage

<!-- Basic -->
<table class="pk-table">
  <thead><tr><th>Name</th><th>Status</th></tr></thead>
  <tbody>
    <tr><td>Alice</td><td>Active</td></tr>
  </tbody>
</table>

<!-- Striped + Hover + Primary header -->
<table class="pk-table pk-table-primary pk-table-striped pk-table-hover">...</table>

<!-- Responsive -->
<div class="pk-table-responsive">
  <table class="pk-table">...</table>
</div>
