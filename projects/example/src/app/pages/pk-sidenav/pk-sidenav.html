Installation

import { PkSidenav } from 'ngx-pk-ui';
import type { PkSidenavGroup, PkSidenavItem } from 'ngx-pk-ui';

@Component({
  imports: [PkSidenav],
})
export class MyComponent {
  activeKey = signal('home');

  readonly groups: PkSidenavGroup[] = [
    {
      heading: 'Main',
      items: [
        { key: 'home',     label: 'Home',          icon: 'home' },
        { key: 'notif',    label: 'Notifications', icon: 'notifications', badge: 5 },
        { key: 'analytics',label: 'Analytics',     icon: 'bar_chart' },
      ],
    },
    {
      heading: 'Settings',
      items: [
        {
          key: 'reports', label: 'Reports', icon: 'folder',
          children: [
            { key: 'r-all',  label: 'All reports', icon: 'list' },
            { key: 'r-saved',label: 'Saved',       icon: 'bookmark' },
          ],
        },
        { key: 'settings', label: 'Settings', icon: 'settings' },
      ],
    },
  ];

  onItemClick(item: PkSidenavItem) { this.activeKey.set(item.key); }
}

Usage

<pk-sidenav
  [groups]="groups"
  theme="dark"
  mode="auto"
  position="left"
  [activeKey]="activeKey()"
  (itemClick)="onItemClick($event)"
>
  <!-- optional: brand header slot -->
  <div pkSidenavHeader style="display:flex;align-items:center;gap:10px;font-weight:700;">
    <span>Logo</span>
    <span class="pk-snv-hide-on-icon">My App</span>
  </div>

  <!-- optional: user footer slot -->
  <div pkSidenavUser style="display:flex;align-items:center;gap:10px;">
    <img src="avatar.jpg" style="width:34px;height:34px;border-radius:50%;" />
    <span class="pk-snv-hide-on-icon">John Doe</span>
  </div>
</pk-sidenav>

<!-- custom theme example -->
<pk-sidenav
  theme="custom"
  [themeConfig]="{ bg: '#1e1b4b', color: '#c7d2fe', activeColor: '#a5b4fc', activeBg: 'rgba(165,180,252,0.12)', activeBorder: '#a5b4fc' }"
  [groups]="groups"
  [activeKey]="activeKey()"
  (itemClick)="onItemClick($event)"
/>
