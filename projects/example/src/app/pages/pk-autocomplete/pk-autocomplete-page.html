Usage

import { FormsModule } from '@angular/forms';
import { PkAutocompleteComponent, type AutocompleteOption } from 'ngx-pk-ui';

@Component({
  imports: [FormsModule, PkAutocompleteComponent],
})
export class AnyPage {
  options: AutocompleteOption[] = [
    { label: 'Bangkok', value: 'bangkok' },
    { label: 'Phuket', value: 'phuket' },
  ];

  selected: string | null = null;
}

<pk-autocomplete
  [options]="options"
  placeholder="Search"
  [(ngModel)]="selected"
/>

API

Input Type Default Description
options AutocompleteOption[] [] Local options data source
fetchFn (term) => Promise<AutocompleteOption[]> null Async fetch function for remote search
placeholder string 'ค้นหา...' Input placeholder text
minChars number 1 Minimum characters before async fetch
debounceTime number 300 Debounce time for async searching
disabled boolean false Disable component interaction
[(ngModel)] any — Two-way selected value binding
