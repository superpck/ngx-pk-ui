<!-- Method 1: inline style attribute -->
<pk-tabs style="--pk-tabs-active-color: #0f766e; --pk-tabs-active-bg: #ccfbf1;"> ... </pk-tabs>

<!-- Method 2: [customStyle] object binding -->
<pk-tabs [customStyle]="{ '--pk-tabs-active-color': '#0f766e', '--pk-tabs-active-bg': '#ccfbf1' }">
 ... 
</pk-tabs>

<!-- Method 3: global CSS / :root -->
pk-tabs {
  --pk-tabs-active-color: #0f766e;
  --pk-tabs-active-bg: #ccfbf1;
  --pk-tabs-hover-bg: #e0fdf4;
  --pk-tabs-header-bg: #f0fdfa;
  --pk-tabs-header-border: #99f6e4;
  --pk-tabs-content-bg: #f8fffd;
}

Installation

import { PkTabsModule } from 'ngx-pk-ui';

@Component({
  imports: [PkTabsModule],
  // ...
})

Usage

<pk-tabs>
  <pk-tab>
    <pk-tab-title>Profile</pk-tab-title>
    <pk-tab-body>
      <p>Profile content here</p>
    </pk-tab-body>
  </pk-tab>
  <pk-tab>
    <pk-tab-title>Settings</pk-tab-title>
    <pk-tab-body>
      <p>Settings content here</p>
    </pk-tab-body>
  </pk-tab>
  <pk-tab [disabled]="true">
    <pk-tab-title>Disabled</pk-tab-title>
    <pk-tab-body>
      <p>Never shown</p>
    </pk-tab-body>
  </pk-tab>
</pk-tabs>

API — pk-tabs (PkTabsComponent)

Inputs

Input Type Default Description
overflow 'scrollx' | 'newline' 'scrollx' Tab header overflow behavior — horizontal scroll or wrap to new line
activeColor string '' Shortcut to set active-tab accent color (any CSS color). Auto-derives hover and active-bg tints.
customClass string '' Extra CSS class on the container element
customStyle { [key: string]: any } {} Inline styles on the container — also accepts --pk-tabs-* CSS variables for full color control
style string '' Raw style string on the container and header

Outputs

Output Emits Description
(onSelectTab) number Emits the zero-based index of the newly selected tab

CSS Custom Properties

Variable Default Description
--pk-tabs-active-color #0d6efd Active tab underline color
--pk-tabs-active-bg #0066ff1e Active tab background tint
--pk-tabs-hover-bg rgba(0,0,0,0.063) Tab hover background
--pk-tabs-header-bg #f8f9fa Tab header bar background
--pk-tabs-header-border #dee2e6 Header bottom border color
--pk-tabs-content-bg #ffffff Content area background

API — pk-tab (PkTabComponent)

Inputs

Input Type Default Description
active boolean false Set this tab as the initially active tab
disabled boolean false Prevent the tab from being selected
customClass string '' Extra CSS class on the tab panel
customStyle { [key: string]: any } {} Inline styles on the tab panel
style string '' Raw style string on the tab panel

Content projection: place <pk-tab-title> and <pk-tab-body> inside <pk-tab>.

API — pk-tab-title (PkTabTitleComponent)

Inputs

Input Type Default Description
customClass string '' Extra CSS class on the title element
customStyle { [key: string]: string } {} Inline styles on the title element
style string '' Raw style string on the title element

Outputs

Output Emits Description
(tabClick) void Emits when the title is clicked

API — pk-tab-body (PkTabBodyComponent)

Inputs

Input Type Default Description
customClass string '' Extra CSS class on the body element
customStyle { [key: string]: any } {} Inline styles on the body element
