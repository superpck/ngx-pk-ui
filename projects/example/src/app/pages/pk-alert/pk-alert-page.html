pk-alert

Promise-based modal dialogs — alert, confirm, and input (string / number / date / boolean).

Installation

<!-- Add once to your root component template -->
<pk-alert />

Usage

import { PkAlertService } from 'ngx-pk-ui';

@Component({ ... })
export class MyComponent {
  alert = inject(PkAlertService);

  async examples() {
    // Notifications (resolves when user clicks OK)
    await this.alert.success('Saved successfully!');
    await this.alert.warn('This may cause issues.');
    await this.alert.error('Operation failed.');

    // Confirm (resolves true/false)
    const yes = await this.alert.confirm('Delete?', 'Confirm');
    if (yes) { /* proceed */ }

    // Input dialogs
    const name = await this.alert.input('Your name:', 'string', 'Name', 'Alice');
    const age  = await this.alert.input('Your age:',  'number', 'Age', 25);
    const dob  = await this.alert.input('Pick date:', 'date');
    const ok   = await this.alert.input('Agree?',     'boolean');
  }
}

API

MethodReturnsDescription
success(message, title?)Promise<void>Info dialog with green icon
warn(message, title?)Promise<void>Warning dialog with orange icon
error(message, title?)Promise<void>Error dialog with red icon
confirm(message, title?, okLabel?, cancelLabel?)Promise<boolean>Confirm with OK/Cancel
input(message, type, title?, default?)Promise<string|number|boolean|null>Inline input; null if cancelled

type values: 'string' 'number' 'date' 'boolean'
